function salvarChecklist() { // ... (código existente) const respostas = []; const questoes = document.querySelectorAll('#checklist-questoes .mb-2'); const respostasVistas = new Set(); questoes.forEach((questao, index) => { const descricao = questao.querySelector('p')?.textContent || ''; const conformidade = questao.querySelector('select')?.value || ''; const observacoes = questao.querySelector('textarea')?.value || ''; // Crie uma chave única para cada resposta const respostaKey = `${descricao}-${conformidade}`; if (!respostasVistas.has(respostaKey)) { respostas.push({ id: index + 1, descricao: descricao, conformidade: conformidade, observacoes: observacoes }); respostasVistas.add(respostaKey); } }); // ... (resto do código) }
